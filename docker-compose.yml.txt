version: '3.8'

services:
  postgres:
    image: postgres
    restart: always
    environment:
      POSTGRES_USER: user
      POSTGRES_PASSWORD: password
      POSTGRES_DB: dbname
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data

  api:
    build:
      context: .
      dockerfile: Api/Dockerfile
    restart: always
    ports:
      - "5000:80"
    depends_on:
      - postgres

  frontend:
    build:
      context: .
      dockerfile: Frontend/Dockerfile
    restart: always
    ports:
      - "4200:4200"
